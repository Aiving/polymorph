47struct RoundedCorner {
48    p0: Point,
49    p1: Point,
50    p2: Point,
51
52    d1: Vector,
53    d2: Vector,
54
55    corner_radius: f32,
56    smoothing: f32,
57    expected_round_cut: f32,
58

60}

62const DISTANCE_EPSILON: f32 = 1e-4f32;
63
64impl RoundedCorner {
65    pub fn new(p0: Point, p1: Point, p2: Point, rounding: Option<CornerRounding>) -> Self {
66        let mut d1 = Vector::zero();
67        let mut d2 = Vector::zero();
68        let mut corner_radius = 0.0;
69        let mut smoothing = 0.0;
70        let mut expected_round_cut = 0.0;
71
72        let v01 = p0 - p1;
73        let v21 = p2 - p1;
74        let d01 = v01.length();
75        let d21 = v21.length();
76
77        if d01 > 0.0 && d21 > 0.0 {
78            d1 = v01 / d01;
79            d2 = v21 / d21;
80
81            corner_radius = rounding.map_or(0.0, |r| r.radius);
82            smoothing = rounding.map_or(0.0, |r| r.smoothing);
83
84            // cosine of angle at p1 is dot product of unit vectors to the other two
85            // vertices
86            let cos_angle = d1.dot(d2);
87
88            // identity: sin^2 + cos^2 = 1
89            // sinAngle gives us the intersection
90            let sin_angle = cos_angle.mul_add(-cos_angle, 1.0).sqrt();
91
92            // How much we need to cut, as measured on a side, to get the required radius
93            // calculating where the rounding circle hits the edge
94            // This uses the identity of tan(A/2) = sinA/(1 + cosA), where tan(A/2) =
95            // radius/cut
96
97            if sin_angle > 1e-3 {
98                expected_round_cut = corner_radius * (cos_angle + 1.0) / sin_angle;
99            }
100        }
101
102        Self {
103            p0,
104            p1,
105            p2,
106            d1,
107            d2,
108            corner_radius,
109            smoothing,
110            expected_round_cut,
111            center: Point::zero(),
112        }
113    }
114
115    pub const fn expected_cut(&self) -> f32 {
116        (1.0 + self.smoothing) * self.expected_round_cut
117    }
118
119    fn calculate_actual_smoothing_value(&self, allowed_cut: f32) -> f32 {
120        if allowed_cut > self.expected_cut() {
121            self.smoothing
122        } else if allowed_cut > self.expected_round_cut {
123            self.smoothing * (allowed_cut - self.expected_round_cut) / (self.expected_cut() - self.expected_round_cut)
124        } else {
125            0.0
126        }
127    }
128
129    fn line_intersection(p0: Point, d0: Point, p1: Point, d1: Point) -> Option<Point> {
130        let rotated_d1 = d1.rotate90();
131        let den = d0.to_vector().dot(rotated_d1.to_vector());
132
133        if den.abs() < DISTANCE_EPSILON {
134            return None;
135        }
136
137        let num = (p1 - p0).dot(rotated_d1.to_vector());
138
139        // Also check the relative value. This is equivalent to abs(den/num) <
140        // DISTANCE_EPSILON, but avoid doing a division
141        if den.abs() < DISTANCE_EPSILON * num.abs() {
142            None
143        } else {
144            let k = num / den;
145
146            Some(p0 + (d0 * k).to_vector())
147        }
148    }

150    #[allow(clippy::too_many_arguments)]
151    fn compute_flanking_curve(
152        actual_round_cut: f32,
153        actual_smoothing_values: f32,
154        corner: Point,
155        side_start: Point,
156        circle_segment_intersection: Point,
157        other_circle_segment_intersection: Point,
158        circle_center: Point,
159        actual_r: f32,
160    ) -> Cubic {
161        // side_start is the anchor, 'anchor' is actual control point
162        let side_direction = (side_start - corner).to_point().get_direction();
163        let curve_start = corner + (side_direction * actual_round_cut * (1.0 + actual_smoothing_values)).to_vector();
164        // We use an approximation to cut a part of the circle section proportional to 1
165        // - smooth, When smooth = 0, we take the full section, when smooth = 1,
166        // we take nothing. TODO: revisit this, it can be problematic as it
167        // approaches 180 degrees
168        let p = circle_segment_intersection.lerp(
169            (circle_segment_intersection + other_circle_segment_intersection.to_vector()) / 2.0,
170            actual_smoothing_values,
171        );
172        // The flanking curve ends on the circle
173        let curve_end = circle_center + (p - circle_center).normalize() * actual_r;
174        // The anchor on the circle segment side is in the intersection between the
175        // tangent to the circle in the circle/flanking curve boundary and the
176        // linear segment.
177        let circle_tangent = (curve_end - circle_center).to_point().rotate90();
178        let anchor_end = Self::line_intersection(side_start, side_direction, curve_end, circle_tangent).unwrap_or(circle_segment_intersection);
179        // From what remains, we pick a point for the start anchor.
180        // 2/3 seems to come from design tools?
181        let anchor_start = (curve_start + (anchor_end * 2.0).to_vector()) / 3.0;
182
183        Cubic::new(curve_start, anchor_start, anchor_end, curve_end)
184    }
185
186    pub fn get_cubics(&mut self, allowed_cut0: f32, allowed_cut1: f32) -> Vec<Cubic> {
187        // We use the minimum of both cuts to determine the radius, but if there is more
188        // space in one side we can use it for smoothing.
189        let allowed_cut = allowed_cut0.min(allowed_cut1);
190
191        // Nothing to do, just use lines, or a point
192        if self.expected_round_cut < DISTANCE_EPSILON || allowed_cut < DISTANCE_EPSILON || self.corner_radius < DISTANCE_EPSILON {
193            self.center = self.p1;
194
195            return vec![Cubic::straight_line(self.p1, self.p1)];
196        }
197
198        // How much of the cut is required for the rounding part.
199        let actual_round_cut = allowed_cut.min(self.expected_round_cut);
200        // We have two smoothing values, one for each side of the vertex
201        // Space is used for rounding values first. If there is space left over, then we
202        // apply smoothing, if it was requested
203        let actual_smoothing0 = self.calculate_actual_smoothing_value(allowed_cut0);
204        let actual_smoothing1 = self.calculate_actual_smoothing_value(allowed_cut1);
205        // Scale the radius if needed
206        let actual_r = self.corner_radius * actual_round_cut / self.expected_round_cut;
207        // Distance from the corner (p1) to the center
208        let center_distance = actual_r.hypot(actual_round_cut);
209
210        // Center of the arc we will use for rounding
211        self.center = self.p1 + (((self.d1 + self.d2) / 2.0).get_direction() * center_distance);
212
213        let circle_intersection0 = self.p1 + (self.d1 * actual_round_cut);
214        let circle_intersection2 = self.p1 + (self.d2 * actual_round_cut);
215
216        let flanking0 = Self::compute_flanking_curve(
217            actual_round_cut,
218            actual_smoothing0,
219            self.p1,
220            self.p0,
221            circle_intersection0,
222            circle_intersection2,
223            self.center,
224            actual_r,
225        );
226
227        let flanking2 = Self::compute_flanking_curve(
228            actual_round_cut,
229            actual_smoothing1,
230            self.p1,
231            self.p2,
232            circle_intersection2,
233            circle_intersection0,
234            self.center,
235            actual_r,
236        )
237        .reversed();
238
239        let flanking1 = Cubic::circular_arc(self.center, flanking0.anchor1(), flanking2.anchor0());
240
241        vec![flanking0, flanking1, flanking2]
242    }

245/// [`RoundedPolygon`] allows simple construction of polygonal shapes with
246/// optional rounding at the vertices.
247///
248/// Polygons can be constructed with either the number of vertices desired or an
249/// ordered list of vertices.
250#[derive(Debug, Clone, PartialEq)]
251pub struct RoundedPolygon {
252    pub features: Vec<Feature>,
253    pub center: Point,
254    /// A flattened version of the [`Feature`]s.
255    pub cubics: Vec<Cubic>,
256}
257
258impl RoundedPolygon {
259    pub fn new(features: Vec<Feature>, center: Point) -> Self {
260        let mut cubics = Vec::new();
261
262        // The first/last mechanism here ensures that the final anchor point in the
263        // shape exactly matches the first anchor point. There can be rendering
264        // artifacts introduced by those points being slightly off, even by much
265        // less than a pixel
266        let mut first_cubic: Option<Cubic> = None;
267        let mut last_cubic: Option<Cubic> = None;
268        let mut first_feature_split_start: Option<Vec<Cubic>> = None;
269        let mut first_feature_split_end: Option<Vec<Cubic>> = None;
270
271        if !features.is_empty() && features[0].cubics.len() == 3 {
272            let center_cubic = features[0].cubics[1];
273            let (start, end) = center_cubic.split(0.5);
274
275            first_feature_split_start.replace(vec![features[0].cubics[0], start]);
276            first_feature_split_end.replace(vec![end, features[0].cubics[2]]);
277        }
278
279        // iterating one past the features list size allows us to insert the initial
280        // split cubic if it exists
281        for i in 0..=features.len() {
282            let feature_cubics = if i == 0 && first_feature_split_end.is_some() {
283                let Some(cubics) = &first_feature_split_end else { unreachable!() };
284
285                cubics
286            } else if i == features.len() {
287                if let Some(cubics) = &first_feature_split_start {
288                    cubics
289                } else {
290                    break;
291                }
292            } else {
293                &features[i].cubics
294            };
295
296            for &cubic in feature_cubics {
297                // Skip zero-length curves; they add nothing and can trigger rendering artifacts
298                if !cubic.zero_length() {
299                    if let Some(last_cubic) = last_cubic {
300                        cubics.push(last_cubic);
301                    }
302
303                    last_cubic.replace(cubic);
304
305                    if first_cubic.is_none() {
306                        first_cubic.replace(cubic);
307                    }
308                } else if let Some(last_cubic) = last_cubic.as_mut() {
309                    // Dropping several zero-ish length curves in a row can lead to
310                    // enough discontinuity to throw an exception later, even though the
311                    // distances are quite small. Account for that by making the last
312                    // cubic use the latest anchor point, always.
313                    last_cubic.points[3] = cubic.anchor1();
314                }
315            }
316        }
317
318        if let (Some(last_cubic), Some(first_cubic)) = (last_cubic, first_cubic) {
319            cubics.push(Cubic::new(
320                last_cubic.anchor0(),
321                last_cubic.control0(),
322                last_cubic.control1(),
323                first_cubic.anchor0(),
324            ));
325        } else {
326            // Empty / 0-sized polygon.
327            cubics.push(Cubic::new(center, center, center, center));
328        }
329
330        Self { features, center, cubics }
331    }

333    pub fn from_features(features: Vec<Feature>, center: Option<Point>) -> Self {
334        let center = center.unwrap_or_else(|| Point::splat(f32::NAN));
335        let vertices = features
336            .iter()
337            .flat_map(|feature| feature.cubics.iter().map(Cubic::anchor0))
338            .collect::<Vec<_>>();
339
340        let center = if center.x.is_nan() || center.y.is_nan() {
341            let center_from_verts = center_from_vertices(&vertices);
342
343            Point::new(
344                if center.x.is_nan() { center_from_verts.x } else { center.x },
345                if center.y.is_nan() { center_from_verts.y } else { center.y },
346            )
347        } else {
348            center
349        };
350
351        Self::new(features, center)
352    }

439    /// # Panics
440    ///
441    /// May panic if:
442    /// - The polygon contains fewer than 3 vertices
443    /// - The `per_vertex_rounding` is not empty, but its size does not
444    ///   correspond to the number of vertices in the polygon
445    pub fn from_vertices(vertices: &[Point], rounding: CornerRounding, per_vertex_rounding: &[CornerRounding], center: Point) -> Self {
446        assert!(vertices.len() >= 3, "Polygons must have at least 3 vertices");
447        assert!(
448            per_vertex_rounding.is_empty() || per_vertex_rounding.len() == vertices.len(),
449            "per_vertex_rounding array should be either empty or the same size as the number of vertices"
450        );
451
452        let mut corners = <Vec<Vec<Cubic>>>::new();
453        let n = vertices.len();
454        let mut rounded_corners = <Vec<RoundedCorner>>::new();
455
456        for i in 0..n {
457            rounded_corners.push(RoundedCorner::new(
458                vertices[(i + n - 1) % n],
459                vertices[i],
460                vertices[(i + 1) % n],
461                Some(per_vertex_rounding.get(i).copied().unwrap_or(rounding)),
462            ));
463        }
464
465        let cut_adjusts = (0..n)
466            .map(|ix| {
467                let expected_round_cut = rounded_corners[ix].expected_round_cut + rounded_corners[(ix + 1) % n].expected_round_cut;
468                let expected_cut = rounded_corners[ix].expected_cut() + rounded_corners[(ix + 1) % n].expected_cut();
469                let vtx = vertices[ix];
470                let next_vtx = vertices[(ix + 1) % n];
471                let side_size = (vtx.x - next_vtx.x).hypot(vtx.y - next_vtx.y);
472
473                // Check expected_round_cut first, and ensure we fulfill rounding needs first
474                // for both corners before using space for smoothing
475                if expected_round_cut > side_size {
476                    // Not enough room for fully rounding, see how much we can actually do.
477                    (side_size / expected_round_cut, 0.0)
478                } else if expected_cut > side_size {
479                    // We can do full rounding, but not full smoothing.
480                    (1.0, (side_size - expected_round_cut) / (expected_cut - expected_round_cut))
481                } else {
482                    // There is enough room for rounding & smoothing.
483                    (1.0, 1.0)
484                }
485            })
486            .collect::<Vec<_>>();
487
488        // Create and store list of beziers for each potentially rounded corner
489        for i in 0..n {
490            // allowed_cuts[0] is for the side from the previous corner to this one,
491            // allowed_cuts[1] is for the side from this corner to the next one.
492            let mut allowed_cuts = [0.0; 2];
493
494            for delta in 0..=1 {
495                let (round_cut_ratio, cut_ratio) = cut_adjusts[(i + n - 1 + delta) % n];
496
497                allowed_cuts[delta] = rounded_corners[i].expected_round_cut.mul_add(
498                    round_cut_ratio,
499                    (rounded_corners[i].expected_cut() - rounded_corners[i].expected_round_cut) * cut_ratio,
500                );
501            }
502
503            let corner = rounded_corners[i].get_cubics(allowed_cuts[0], allowed_cuts[1]);
504
505            corners.push(corner);
506        }
507
508        let mut temp_features = <Vec<Feature>>::new();
509
510        for i in 0..n {
511            // Note that these indices are for pairs of values (points), they need to be
512            // doubled to access the xy values in the vertices float array
513            let prev_vtx_index = (i + n - 1) % n;
514            let next_vtx_index = (i + 1) % n;
515            let convex = vertices[prev_vtx_index].is_convex(vertices[i], vertices[next_vtx_index]);
516
517            temp_features.push(Feature::corner(corners[i].clone(), convex));
518            temp_features.push(Feature::edge(vec![Cubic::straight_line(
519                corners[i].last().unwrap().anchor1(),
520                corners[(i + 1) % n].first().unwrap().anchor0(),
521            )]));
522        }
523
524        let center = if center.x == f32::MIN || center.y == f32::MIN {
525            center_from_vertices(vertices)
526        } else {
527            center
528        };
529
530        Self::new(temp_features, center)
531    }

543    /// Returns an axis-aligned bounding box describing bounds of the polygon.
544    ///
545    /// If `approximate` is `true`, a fast but sometimes inaccurate algorithm is
546    /// used to calculate AABB of cubics. Otherwise, it finds the derivative,
547    /// which is a quadratic Bzier curve, and then solves the equation for `t`
548    /// using the quadratic formula.
549    pub fn aabb(&self, approximate: bool) -> Aabb {
550        let mut aabb = Aabb::new(Point::splat(f32::MAX), Point::splat(f32::MIN));
551
552        for cubic in &self.cubics {
553            let cubic_aabb = cubic.aabb(approximate);
554
555            aabb = Aabb {
556                min: aabb.min.min(cubic_aabb.min),
557                max: aabb.max.max(cubic_aabb.max),
558            };
559        }
560
561        aabb
562    }

564    /// Moves and resizes [`RoundedPolygon`], so it's completely inside the 0x0
565    /// -> 1x1 square, centered if there extra space in one direction.
566    #[must_use]
567    pub fn normalized(self) -> Self {
568        let bounds = self.aabb(true);
569        let size = bounds.size();
570
571        let max_side = size.width.max(size.height);
572
573        let offset = ((Point::splat(max_side) - size) / 2.0 - bounds.min).to_point();
574
575        self.transformed(|point| (point + offset.to_vector()) / max_side)
576    }

594fn center_from_vertices(vertices: &[Point]) -> Point {
595    let mut cumulative = Point::zero();
596
597    for vertice in vertices {
598        cumulative += vertice.to_vector();
599    }
600
601    cumulative / vertices.len() as f32
602}

604#[allow(clippy::manual_is_multiple_of)] // For MSRV compability
605fn custom_polygon(points: &[RoundedPoint], repeats: usize, center: Option<Point>, mirroring: bool) -> RoundedPolygon {
606    let center = center.unwrap_or(Point::new(0.5, 0.5));
607    let mut actual_points = Vec::new();
608
609    if mirroring {
610        let angles = points
611            .iter()
612            .map(|it| (it.offset - center).angle_from_x_axis().to_degrees())
613            .collect::<Vec<_>>();
614        let distances = points.iter().map(|it| (it.offset - center).length()).collect::<Vec<_>>();
615        let actual_repeats = repeats * 2;
616        let section_angle = 360.0 / actual_repeats as f32;
617
618        for iteration in 0..actual_repeats {
619            for index in 0..points.len() {
620                let i = if iteration % 2 == 0 { index } else { points.len() - index - 1 };
621
622                if i > 0 || iteration % 2 == 0 {
623                    let angle = section_angle
624                        .mul_add(
625                            iteration as f32,
626                            if iteration % 2 == 0 {
627                                angles[i]
628                            } else {
629                                2f32.mul_add(angles[0], section_angle - angles[i])
630                            },
631                        )
632                        .to_radians();
633
634                    let final_point = Point::new(angle.cos(), angle.sin()) * distances[i] + center.to_vector();
635
636                    actual_points.push(RoundedPoint {
637                        offset: final_point,
638                        rounding: points[i].rounding,
639                    });
640                }
641            }
642        }
643    } else {
644        let size = points.len();
645
646        for it in 0..size * repeats {
647            let point = points[it % size].offset.rotated((it / size) as f32 * 360.0 / repeats as f32, center);
648
649            actual_points.push(RoundedPoint {
650                offset: point,
651                rounding: points[it % size].rounding,
652            });
653        }
654    }
655
656    RoundedPolygon::from_vertices(
657        &actual_points.iter().map(|p| p.offset).collect::<Vec<_>>(),
658        CornerRounding::UNROUNDED,
659        &actual_points.iter().map(|p| p.rounding).collect::<Vec<_>>(),
660        center,
661    )
662}
663
664fn vertices_from_count(count: usize, radius: f32, center: Point) -> Vec<Point> {
665    (0..count)
666        .map(|i| center + radial_to_cartesian(radius, f32::consts::PI / count as f32 * 2.0 * i as f32))
667        .collect()
668}
