83/// Provides a more convenient and familiar way to create certain shapes using
84/// builder pattern, helping to avoid functions with hellish `Option<T>`
85/// arguments.
86///
87/// Currently supports the creation of the following shapes:
88/// - Rectangles
89/// - Circles
90/// - Stars
91/// - Pills
92/// - Pill stars
93pub struct RoundedPolygonBuilder<T> {
94    pub(crate) data: T,
95    pub(crate) center: Point,
96    pub(crate) rounding: CornerRounding,
97    pub(crate) per_vertex_rounding: Vec<CornerRounding>,
98}

186impl RoundedPolygonBuilder<Rectangle> {
187    /// Sets the common [`CornerRounding`] for each of the rectangle corners.
188    #[must_use]
189    pub const fn with_rounding(mut self, rounding: CornerRounding) -> Self {
190        self.rounding = rounding;
191
192        self
193    }
194
195    /// Sets [`CornerRounding`] for each individual corner. The size of the
196    /// iterator must be equal to 4 for four rectangle corners.
197    #[must_use]
198    pub fn with_rounding_per_vertex(mut self, corners: [CornerRounding; 4]) -> Self {
199        self.per_vertex_rounding = corners.into();
200
201        self
202    }
203
204    /// Consuming itself, it creates and returns [`RoundedPolygon`], which forms
205    /// a rectangle.
206    pub fn build(self) -> RoundedPolygon {
207        let [left, top] = (self.center - self.data.size / 2.0).to_array();
208        let [right, bottom] = (self.center + self.data.size / 2.0).to_array();
209
210        let vertices = [
211            Point::new(right, bottom),
212            Point::new(left, bottom),
213            Point::new(left, top),
214            Point::new(right, top),
215        ];
216
217        RoundedPolygon::from_vertices(&vertices, self.rounding, &self.per_vertex_rounding, self.center)
218    }

221impl RoundedPolygonBuilder<Star> {
222    /// Set inner radius for the resulting star shape, which must be greater
223    /// than 0 and less than or equal to radius.
224    #[must_use]
225    pub const fn with_inner_radius(mut self, radius: f32) -> Self {
226        self.data.inner_radius = radius;
227
228        self
229    }
230
231    /// Sets the common [`CornerRounding`] for each of the vertices on the inner
232    /// radius.
233    #[must_use]
234    pub const fn with_inner_rounding(mut self, rounding: CornerRounding) -> Self {
235        self.data.inner_rounding.replace(rounding);
236
237        self
238    }
239
240    /// Consuming itself, it creates and returns [`RoundedPolygon`], which forms
241    /// a star.
242    pub fn build(self) -> RoundedPolygon {
243        let vertices = star_vertices_from_num_verts(self.data.vertices_per_radius, self.data.radius, self.data.inner_radius, self.center);
244
245        // Star polygon is just a polygon with all vertices supplied (where we generate
246        // those vertices to be on the inner/outer radii)
247        if !self.per_vertex_rounding.is_empty() {
248            RoundedPolygon::from_vertices(&vertices, self.rounding, &self.per_vertex_rounding, self.center)
249        } else if let Some(inner_rounding) = self.data.inner_rounding {
250            // If no per-vertex rounding supplied and caller asked for inner
251            // rounding, create per-vertex rounding list based on
252            // supplied outer/inner rounding parameters
253            RoundedPolygon::from_vertices(
254                &vertices,
255                self.rounding,
256                &(0..self.data.vertices_per_radius)
257                    .flat_map(|_| [self.rounding, inner_rounding])
258                    .collect::<Vec<_>>(),
259                self.center,
260            )
261        } else {
262            RoundedPolygon::from_vertices(&vertices, self.rounding, &[], self.center)
263        }
264    }
265}

267impl RoundedPolygonBuilder<Pill> {
268    /// Sets the amount by which the arc is "smoothed" by extending the curve
269    /// from the circular arc on each endcap to the edge between the endcaps.
270    #[must_use]
271    pub const fn with_smoothing(mut self, smoothing: f32) -> Self {
272        self.data.smoothing = smoothing;
273
274        self
275    }
276
277    /// Consuming itself, it creates and returns [`RoundedPolygon`], which forms
278    /// a pill.
279    pub fn build(self) -> RoundedPolygon {
280        let half_size = self.data.size / 2.0;
281
282        RoundedPolygon::from_vertices(
283            &[
284                self.center + half_size,
285                self.center + Vector::new(-half_size.width, half_size.height),
286                self.center - half_size,
287                self.center + Vector::new(half_size.width, -half_size.height),
288            ],
289            CornerRounding::smoothed(half_size.width.min(half_size.height), self.data.smoothing),
290            &[],
291            self.center,
292        )
293    }
294}

374fn pill_star_vertices_from_num_verts(
375    num_vertices_per_radius: usize,
376    size: Size,
377    inner_radius: f32,
378    vertex_spacing: f32,
379    start_location: f32,
380    center: Point,
381) -> Vec<Point> {
382    // The general approach here is to get the perimeter of the underlying pill
383    // outline, then the t value for each vertex as we walk that perimeter. This
384    // tells us where on the outline to place that vertex, then we figure out
385    // where to place the vertex depending on which "section" it is in. The
386    // possible sections are the vertical edges on the sides, the circular
387    // sections on all four corners, or the horizontal edges on the top and
388    // bottom. Note that either the vertical or horizontal edges will be
389    // of length zero (whichever dimension is smaller gets only circular curvature
390    // for the pill shape).
391    let endcap_radius = size.width.min(size.height);
392    let v_seg_len = (size.height - size.width).max(0.0);
393    let h_seg_len = (size.width - size.height).max(0.0);
394    let v_seg_half = v_seg_len / 2.0;
395    let h_seg_half = h_seg_len / 2.0;
396    // vertexSpacing is used to position the vertices on the end caps. The caller
397    // has the choice of spacing the inner (0) or outer (1) vertices like those
398    // along the edges, causing the other vertices to be either further apart
399    // (0) or closer (1). The default is .5, which averages things. The
400    // magnitude of the inner and rounding parameters may cause the caller
401    // to want a different value.
402    let circle_perimeter = f32::consts::PI * 2.0 * endcap_radius * inner_radius.mul_add(1.0 - vertex_spacing, 1.0 * vertex_spacing);
403    // perimeter is circle perimeter plus horizontal and vertical sections of inner
404    // rectangle, whether either (or even both) might be of length zero.
405    let perimeter = 2.0f32.mul_add(h_seg_len, 2.0 * v_seg_len) + circle_perimeter;
406
407    // The sections array holds the t start values of that part of the outline. We
408    // use these to determine which section a given vertex lies in, based on its
409    // t value, as well as where in that section it lies.
410    let mut sections = [0.0; 11];
411
412    sections[1] = v_seg_len / 2.0;
413    sections[2] = sections[1] + circle_perimeter / 4.0;
414    sections[3] = sections[2] + h_seg_len;
415    sections[4] = sections[3] + circle_perimeter / 4.0;
416    sections[5] = sections[4] + v_seg_len;
417    sections[6] = sections[5] + circle_perimeter / 4.0;
418    sections[7] = sections[6] + h_seg_len;
419    sections[8] = sections[7] + circle_perimeter / 4.0;
420    sections[9] = sections[8] + v_seg_len / 2.0;
421    sections[10] = perimeter;
422
423    // "t" is the length along the entire pill outline for a given vertex. With
424    // vertices spaced evenly along this contour, we can determine for any
425    // vertex where it should lie.
426    let t_per_vertex = perimeter / (2 * num_vertices_per_radius) as f32;
427    // separate iteration for inner vs outer, unlike the other shapes, because
428    // the vertices can lie in different quadrants so each needs their own
429    // calculation
430    let mut inner = false;
431    // Increment section index as we walk around the pill contour with our
432    // increasing t values
433    let mut curr_sec_index = 0;
434    // secStart/End are used to determine how far along a given vertex is in the
435    // section in which it lands
436    let mut sec_start = 0.0;
437    let mut sec_end = sections[1];
438    // t value is used to place each vertex. 0 is on the positive x axis,
439    // moving into section 0 to begin with. startLocation, a value from 0 to 1,
440    // varies the location anywhere on the perimeter of the shape
441    let mut t = start_location * perimeter;
442    // The list of vertices to be returned
443    let mut result = Vec::with_capacity(num_vertices_per_radius * 2);
444    let rect_bottom_right = Point::new(h_seg_half, v_seg_half);
445    let rect_bottom_left = Point::new(-h_seg_half, v_seg_half);
446    let rect_top_left = Point::new(-h_seg_half, -v_seg_half);
447    let rect_top_right = Point::new(h_seg_half, -v_seg_half);
448
449    // Each iteration through this loop uses the next t value as we walk around the
450    // shape
451    for _ in 0..num_vertices_per_radius * 2 {
452        // t could start (and end) after 0; extra boundedT logic makes sure it does the
453        // right thing when crossing the boundar past 0 again
454        let bounded_t = t % perimeter;
455
456        if bounded_t < sec_start {
457            curr_sec_index = 0;
458        }
459
460        #[allow(clippy::while_float)]
461        while bounded_t >= sections[(curr_sec_index + 1) % sections.len()] {
462            curr_sec_index = (curr_sec_index + 1) % sections.len();
463            sec_start = sections[curr_sec_index];
464            sec_end = sections[(curr_sec_index + 1) % sections.len()];
465        }
466
467        // find t in section and its proportion of that section's total length
468        let t_in_section = bounded_t - sec_start;
469        let t_proportion = t_in_section / (sec_end - sec_start);
470
471        // The vertex placement in a section varies depending on whether it is on one of
472        // the semicircle endcaps or along one of the straight edges. For the
473        // endcaps, we use tProportion to get the angle along that circular cap
474        // and add the starting angle for that section. For the edges we use a
475        // straight linear calculation given tProportion and the start/end t
476        // values for that edge.
477        let curr_radius = if inner { endcap_radius * inner_radius } else { endcap_radius };
478        let vertex = match curr_sec_index {
479            0 => Point::new(curr_radius, t_proportion * v_seg_half),
480            1 => rect_bottom_right + radial_to_cartesian(curr_radius, t_proportion * f32::consts::PI / 2.0),
481            2 => Point::new(t_proportion.mul_add(-h_seg_len, h_seg_half), curr_radius),
482            3 => rect_bottom_left + radial_to_cartesian(curr_radius, f32::consts::PI / 2.0 + (t_proportion * f32::consts::PI / 2.0)),
483            4 => Point::new(-curr_radius, t_proportion.mul_add(-v_seg_len, v_seg_half)),
484            5 => rect_top_left + radial_to_cartesian(curr_radius, f32::consts::PI + (t_proportion * f32::consts::PI / 2.0)),
485            6 => Point::new(t_proportion.mul_add(h_seg_len, -h_seg_half), -curr_radius),
486            7 => rect_top_right + radial_to_cartesian(curr_radius, f32::consts::PI.mul_add(1.5, t_proportion * f32::consts::PI / 2.0)),
487            // 8
488            _ => Point::new(curr_radius, t_proportion.mul_add(v_seg_half, -v_seg_half)),
489        };
490
491        result.push(vertex + center.to_vector());
492
493        t += t_per_vertex;
494
495        inner = !inner;
496    }
497
498    result
499}

501fn star_vertices_from_num_verts(num_vertices_per_radius: usize, radius: f32, inner_radius: f32, center: Point) -> Vec<Point> {
502    (0..num_vertices_per_radius * 2)
503        .map(|i| {
504            center
505                + radial_to_cartesian(
506                    if i % 2 == 0 { radius } else { inner_radius },
507                    f32::consts::PI / num_vertices_per_radius as f32 * i as f32,
508                )
509        })
510        .collect()
