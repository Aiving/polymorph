8#[derive(Default)]
9struct MappingHelper<'a> {
10    mapping: Vec<(f32, f32)>,
11
12    used_f1: Vec<&'a ProgressableFeature>,
13    used_f2: Vec<&'a ProgressableFeature>,
14}
15
16impl<'a> MappingHelper<'a> {
17    fn add_mapping(&mut self, f1: &'a ProgressableFeature, f2: &'a ProgressableFeature) {
18        // We don't want to map the same feature twice.
19        if self.used_f1.contains(&f1) || self.used_f2.contains(&f2) {
20            return;
21        }
22
23        // Ret is sorted, find where we need to insert this new mapping.
24        if let Err(insertion_index) = self.mapping.binary_search_by(|it| it.0.total_cmp(&f1.progress)) {
25            let n = self.mapping.len();
26
27            // We can always add the first 1 element
28            if n >= 1 {
29                let (before1, before2) = self.mapping[(insertion_index + n - 1) % n];
30                let (after1, after2) = self.mapping[insertion_index % n];
31
32                // We don't want features that are way too close to each other, that will make
33                // the DoubleMapper unstable
34                if progress_distance(f1.progress, before1) < DISTANCE_EPSILON
35                    || progress_distance(f1.progress, after1) < DISTANCE_EPSILON
36                    || progress_distance(f2.progress, before2) < DISTANCE_EPSILON
37                    || progress_distance(f2.progress, after2) < DISTANCE_EPSILON
38                {
39                    return;
40                }
41
42                // When we have 2 or more elements, we need to ensure we are not adding extra
43                // crossings.
44                if n > 1 && !progress_in_range(f2.progress, before2, after2) {
45                    return;
46                }
47            }
48
49            // All good, we can add the mapping.
50            self.mapping.insert(insertion_index, (f1.progress, f2.progress));
51            self.used_f1.push(f1);
52            self.used_f2.push(f2);
53        } else {
54            panic!("There can't be two features with the same progress");
55        }
56    }
57}

59impl DoubleMapper {
60    pub fn from_features(features1: &[ProgressableFeature], features2: &[ProgressableFeature]) -> Self {
61        // We only use corners for this mapping.
62
63        let mut filtered_features1 = Vec::new();
64        let mut filtered_features2 = Vec::new();
65
66        for feature in features1 {
67            if feature.feature.is_corner() {
68                filtered_features1.push(feature);
69            }
70        }
71
72        for feature in features2 {
73            if feature.feature.is_corner() {
74                filtered_features2.push(feature);
75            }
76        }
77
78        let mut distance_vertex_list = Vec::new();
79
80        for f1 in filtered_features1 {
81            for &f2 in &filtered_features2 {
82                let distance = feature_dist_squared(&f1.feature, &f2.feature);
83
84                if distance != f32::MAX {
85                    distance_vertex_list.push(DistanceVertex { distance, f1, f2 });
86                }
87            }
88        }
89
90        distance_vertex_list.sort_by(|a, b| a.distance.total_cmp(&b.distance));
91
92        // Special cases.
93        Self::new(match distance_vertex_list.len() {
94            0 => vec![(0.0, 0.0), (0.5, 0.5)],
95            1 => {
96                let f1 = distance_vertex_list[0].f1.progress;
97                let f2 = distance_vertex_list[0].f2.progress;
98
99                vec![(f1, f2), ((f1 + 0.5) % 1.0, (f2 + 0.5) % 1.0)]
100            }
101            _ => {
102                let mut helper = MappingHelper::default();
103
104                for dv in distance_vertex_list {
105                    helper.add_mapping(dv.f1, dv.f2);
106                }
107
108                helper.mapping
109            }
110        })
111    }
112}

114struct DistanceVertex<'a> {
115    distance: f32,
116    f1: &'a ProgressableFeature,
117    f2: &'a ProgressableFeature,
118}

120fn feature_representative_point(feature: &Feature) -> Point {
121    (feature.cubics[0].anchor0() + feature.cubics[feature.cubics.len() - 1].anchor1().to_vector()) / 2.0
122}
123
124fn feature_dist_squared(f1: &Feature, f2: &Feature) -> f32 {
125    if f1.is_corner_and(|f1_convex| f2.is_corner_and(|f2_convex| f1_convex != f2_convex)) {
126        // Simple hack to force all features to map only to features of the same
127        // concavity, by returning an infinitely large distance in that case
128        f32::MAX
129    } else {
130        (feature_representative_point(f1) - feature_representative_point(f2)).square_length()
131    }
