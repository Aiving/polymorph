3#[derive(Debug, Clone, Copy, PartialEq)]
4pub struct MeasuredCubic {
5    pub(crate) cubic: Cubic,
6    pub(crate) start_outline_progress: f32,
7    pub(crate) end_outline_progress: f32,
8    pub(crate) measured_size: f32,
9}
10
11impl MeasuredCubic {
12    fn new<T: Measurer>(measurer: &T, cubic: Cubic, start_outline_progress: f32, end_outline_progress: f32) -> Self {
13        let size = measurer.measure_cubic(&cubic);
14
15        Self {
16            cubic,
17            start_outline_progress,
18            end_outline_progress,
19            measured_size: size,
20        }
21    }
22
23    pub(crate) fn cut_at_progress<T: Measurer>(self, measurer: &T, cut_outline_progress: f32) -> (Self, Self) {
24        // Floating point errors further up can cause cut_outline_progress to land just
25        // slightly outside of the start/end progress for this cubic, so we limit it
26        // to those bounds to avoid further errors later
27        let bounded_cut_outline_progress = cut_outline_progress.clamp(self.start_outline_progress, self.end_outline_progress);
28        let outline_progress_size = self.end_outline_progress - self.start_outline_progress;
29        let progress_from_start = bounded_cut_outline_progress - self.start_outline_progress;
30
31        let relative_progress = progress_from_start / outline_progress_size;
32        let t = measurer.find_cubic_cut_point(&self.cubic, relative_progress * self.measured_size);
33
34        assert!((0.0..=1.0).contains(&t), "Cubic cut point is expected to be between 0 and 1");
35
36        // c1/c2 are the two new cubics, then we return (MeasuredCubic, MeasuredCubic)
37        // created from them
38        let (c1, c2) = self.cubic.split(t);
39
40        (
41            Self::new(measurer, c1, self.start_outline_progress, bounded_cut_outline_progress),
42            Self::new(measurer, c2, bounded_cut_outline_progress, self.end_outline_progress),
43        )
44    }
45}

53#[derive(Debug, Clone, PartialEq)]
54pub struct MeasuredPolygon<T: Measurer> {
55    pub(crate) measurer: T,
56    pub(crate) cubics: Vec<MeasuredCubic>,
57    pub(crate) features: Vec<ProgressableFeature>,
58}
59
60impl<T: Measurer> MeasuredPolygon<T> {
61    pub fn new(measurer: T, features: Vec<ProgressableFeature>, cubics: &[Cubic], outline_progress: &[f32]) -> Self {
62        let mut measured_cubics = <Vec<MeasuredCubic>>::new();
63        let mut start_outline_progress = 0.0;
64
65        for index in 0..cubics.len() {
66            // Filter out "empty" cubics
67            if (outline_progress[index + 1] - outline_progress[index]) > DISTANCE_EPSILON {
68                measured_cubics.push(MeasuredCubic::new(
69                    &measurer,
70                    cubics[index],
71                    start_outline_progress,
72                    outline_progress[index + 1],
73                ));
74
75                // The next measured cubic will start exactly where this one ends.
76                start_outline_progress = outline_progress[index + 1];
77            }
78        }
79
80        // We could have removed empty cubics at the end. Ensure the last measured cubic
81        // ends at 1f
82        let i = measured_cubics.len() - 1;
83
84        measured_cubics[i].end_outline_progress = 1.0;
85
86        Self {
87            measurer,
88            cubics: measured_cubics,
89            features,
90        }
91    }
92
93    pub fn measure_polygon(measurer: T, polygon: &RoundedPolygon) -> Self {
94        let mut cubics = <Vec<Cubic>>::new();
95        let mut feature_to_cubic = <Vec<(&Feature, usize)>>::new();
96
97        // Get the cubics from the polygon, at the same time, extract the features and
98        // keep a reference to the representative cubic we will use.
99        for feature_index in 0..polygon.features.len() {
100            let feature = &polygon.features[feature_index];
101
102            for cubic_index in 0..feature.cubics.len() {
103                if feature.is_corner() && cubic_index == feature.cubics.len() / 2 {
104                    feature_to_cubic.push((feature, cubics.len()));
105                }
106
107                cubics.push(feature.cubics[cubic_index]);
108            }
109        }
110
111        let measure_results = cubics.iter().fold(vec![0.0], |mut measure, cubic| {
112            measure.push(measure[measure.len() - 1] + measurer.measure_cubic(cubic));
113
114            measure
115        });
116
117        let total_measure = measure_results[measure_results.len() - 1];
118
119        let mut outline_progress = Vec::with_capacity(measure_results.len());
120
121        for measure in measure_results {
122            outline_progress.push(measure / total_measure);
123        }
124
125        let mut features = Vec::new();
126
127        for (feature, ix) in feature_to_cubic {
128            features.push(ProgressableFeature {
129                progress: positive_modulo(outline_progress[ix].midpoint(outline_progress[ix + 1]), 1.0),
130                feature: feature.clone(),
131            });
132        }
133
134        Self::new(measurer, features, &cubics, &outline_progress)
135    }

137    /// # Panics
138    ///
139    /// May panic if `cutting_point` is outside of `0.0..=1.0`.
140    #[must_use]
141    pub fn cut_and_shift(self, cutting_point: f32) -> Self {
142        assert!((0.0..=1.0).contains(&cutting_point), "Cutting point is expected to be between 0 and 1");
143
144        if cutting_point < DISTANCE_EPSILON {
145            return self;
146        }
147
148        // Find the index of cubic we want to cut
149        let target_index = self
150            .cubics
151            .iter()
152            .position(|it| (it.start_outline_progress..=it.end_outline_progress).contains(&cutting_point))
153            .unwrap_or_default();
154
155        // Cut the target cubic.
156        // b1, b2 are two resulting cubics after cut
157        let (b1, b2) = self.cubics[target_index].cut_at_progress(&self.measurer, cutting_point);
158
159        // Construct the list of the cubics we need:
160        // * The second part of the target cubic (after the cut)
161        // * All cubics after the target, until the end + All cubics from the start,
162        //   before the target cubic
163        // * The first part of the target cubic (before the cut)
164        let mut ret_cubics = vec![b2.cubic];
165
166        for i in 1..self.cubics.len() {
167            ret_cubics.push(self.cubics[(i + target_index) % self.cubics.len()].cubic);
168        }
169
170        ret_cubics.push(b1.cubic);
171
172        // Construct the array of outline progress.
173        // For example, if we have 3 cubics with outline progress [0 .. 0.3], [0.3 ..
174        // 0.8] & [0.8 .. 1.0], and we cut + shift at 0.6:
175        // 0. 0123456789 |--|--/-|-|
176        // The outline progresses will start at 0 (the cutting point, that shifs to
177        // 0.0), then 0.8 - 0.6 = 0.2, then 1 - 0.6 = 0.4, then 0.3 - 0.6 + 1 =
178        // 0.7, then 1 (the cutting point again),
179        // all together: (0.0, 0.2, 0.4, 0.7, 1.0)
180        let mut ret_outline_progress = Vec::with_capacity(self.cubics.len() + 2);
181
182        for index in 0..(self.cubics.len() + 2) {
183            ret_outline_progress.push(match index {
184                0 => 0.0,
185                v if v == self.cubics.len() + 1 => 1.0,
186                _ => {
187                    let cubic_index = (target_index + index - 1) % self.cubics.len();
188
189                    positive_modulo(self.cubics[cubic_index].end_outline_progress - cutting_point, 1.0)
190                }
191            });
192        }
193
194        // Shift the feature's outline progress too.
195        let mut new_features = Vec::new();
196
197        for i in 0..self.features.len() {
198            new_features.push(ProgressableFeature {
199                progress: positive_modulo(self.features[i].progress - cutting_point, 1.0),
200                feature: self.features[i].feature.clone(),
201            });
202        }
203
204        // Filter out all empty cubics (i.e. start and end anchor are (almost) the same
205        // point.)
206        Self::new(self.measurer, new_features, &ret_cubics, &ret_outline_progress)
207    }
