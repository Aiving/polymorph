11/// A structure designed to obtain transition cubics between the start and end
12/// [`RoundedPolygon`]s.
13#[derive(Debug, Clone, PartialEq)]
14pub struct Morph {
15    start: RoundedPolygon,
16    end: RoundedPolygon,
17    r#match: Vec<(Cubic, Cubic)>,
18}
19
20impl Morph {
21    /// Matches the [`Cubic`]s of the start and end [`RoundedPolygon`]s, then
22    /// returns an instance of [`Morph`].
23    ///
24    /// # Panics
25    ///
26    /// May panic if not all cubics of both polygons have been matched.
27    pub fn new(start: RoundedPolygon, end: RoundedPolygon) -> Self {
28        let r#match = Self::match_morph(&start, &end);
29
30        Self { start, end, r#match }
31    }
32
33    /// Returns the transition state between the start and end polygons at a
34    /// given `progress` value represented as a list of [`Cubic`]s.
35    pub fn as_cubics(&self, progress: f32) -> Vec<Cubic> {
36        let mut cubics = Vec::new();
37
38        // The first/last mechanism here ensures that the final anchor point in the
39        // shape exactly matches the first anchor point. There can be rendering
40        // artifacts introduced by those points being slightly off, even by much
41        // less than a pixel
42        let mut first_cubic: Option<Cubic> = None;
43        let mut last_cubic: Option<Cubic> = None;
44
45        for i in 0..self.r#match.len() {
46            let cubic = Cubic::from_fn(|it| self.r#match[i].0.points[it].lerp(self.r#match[i].1.points[it], progress));
47
48            if first_cubic.is_none() {
49                first_cubic.replace(cubic);
50            }
51
52            if let Some(cubic) = last_cubic {
53                cubics.push(cubic);
54            }
55
56            last_cubic.replace(cubic);
57        }
58
59        #[allow(clippy::collapsible_if)] // For MSRV compability
60        if let Some(last_cubic) = last_cubic {
61            if let Some(first_cubic) = first_cubic {
62                cubics.push(Cubic::new(
63                    last_cubic.anchor0(),
64                    last_cubic.control0(),
65                    last_cubic.control1(),
66                    first_cubic.anchor0(),
67                ));
68            }
69        }
70
71        cubics
72    }
73
74    /// Returns a path with a drawn transition state (based on the provided
75    /// `progress`). Path is created using the provided `T`, which should
76    /// implement `PathBuilder` and `Default` traits.
77    pub fn as_path<T: PathBuilder + Default>(&self, progress: f32, repeat_path: bool, close_path: bool) -> T::Path {
78        let mut path = T::default();
79
80        self.add_to(progress, &mut path, repeat_path, close_path);
81
82        path.build()
83    }
84
85    /// Adds a transition state (based on the provided `progress`) to the
86    /// `builder`.
87    pub fn add_to<T: PathBuilder>(&self, progress: f32, builder: &mut T, repeat_path: bool, close_path: bool) {
88        let cubics = self.as_cubics(progress);
89
90        add_cubics(builder, repeat_path, close_path, &cubics);
91    }

93    fn match_morph(p1: &RoundedPolygon, p2: &RoundedPolygon) -> Vec<(Cubic, Cubic)> {
94        // Measure polygons, returns lists of measured cubics for each polygon, which
95        // we then use to match start/end curves
96        let measured_polygon1 = MeasuredPolygon::measure_polygon(LengthMeasurer, p1);
97        let measured_polygon2 = MeasuredPolygon::measure_polygon(LengthMeasurer, p2);
98
99        // features1 and 2 will contain the list of corners (just the inner circular
100        // curve) along with the progress at the middle of those corners. These
101        // measurement values are then used to compare and match between the two
102        // polygons
103        let features1 = &measured_polygon1.features;
104        let features2 = &measured_polygon2.features;
105
106        let double_mapper = DoubleMapper::from_features(features1, features2);
107
108        let polygon2_cut_point = double_mapper.map(0.0);
109
110        let bs1 = measured_polygon1;
111        let bs2 = measured_polygon2.cut_and_shift(polygon2_cut_point);
112
113        // Match
114        // Now we can compare the two lists of measured cubics and create a list of
115        // pairs of cubics [ret], which are the start/end curves that represent
116        // the Morph object and the start and end shapes, and which can be
117        // interpolated to animate the between those shapes.
118        let mut ret = <Vec<(Cubic, Cubic)>>::new();
119        // i1/i2 are the indices of the current cubic on the start (1) and end (2)
120        // shapes
121        let mut i1 = 0;
122        let mut i2 = 0;
123        // b1, b2 are the current measured cubic for each polygon
124        let mut b1 = bs1.cubics.get(i1).copied();
125
126        i1 += 1;
127
128        let mut b2 = bs2.cubics.get(i2).copied();
129
130        i2 += 1;
131
132        // Iterate until all curves are accounted for and matched
133        while let (Some(bb1), Some(bb2)) = (b1.take(), b2.take()) {
134            // Progresses are in shape1's perspective
135            // b1a, b2a are ending progress values of current measured cubics in 0..=1 range
136            let b1a = if i1 == bs1.cubics.len() { 1.0 } else { bb1.end_outline_progress };
137            let b2a = if i2 == bs2.cubics.len() {
138                1.0
139            } else {
140                double_mapper.map_back(positive_modulo(bb2.end_outline_progress + polygon2_cut_point, 1.0))
141            };
142            let minb = b1a.min(b2a);
143
144            // min b is the progress at which the curve that ends first ends.
145            // If both curves ends roughly there, no cutting is needed, we have a match.
146            // If one curve extends beyond, we need to cut it.
147            let (seg1, newb1) = if b1a > minb + ANGLE_EPSILON {
148                let (a, b) = bb1.cut_at_progress(&bs1.measurer, minb);
149
150                (a, Some(b))
151            } else {
152                let value = bs1.cubics.get(i1).copied();
153
154                i1 += 1;
155
156                (bb1, value)
157            };
158
159            let (seg2, newb2) = if b2a > minb + ANGLE_EPSILON {
160                let (a, b) = bb2.cut_at_progress(&bs2.measurer, positive_modulo(double_mapper.map(minb) - polygon2_cut_point, 1.0));
161
162                (a, Some(b))
163            } else {
164                let value = bs2.cubics.get(i2).copied();
165
166                i2 += 1;
167
168                (bb2, value)
169            };
170
171            ret.push((seg1.cubic, seg2.cubic));
172
173            b1 = newb1;
174            b2 = newb2;
175        }
176
177        assert!(b1.is_none() && b2.is_none(), "Expected both Polygon's Cubic to be fully matched");
178
179        ret
180    }
